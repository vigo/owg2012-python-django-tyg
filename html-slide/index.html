İlk Gördüğüm Test

                require 'test/unit'
                class TestFunWithStrings < Test::Unit::TestCase
                  def test_palindrome
                    palindromes = ["Satan Oscillate My Metallic Sonatas", "Ekalaka Lake", ""]
                    not_palindromes = ["Abracadabra"]
                    palindromes.each { |pal| assert palindrome?(pal) }
                    not_palindromes.each { |not_pal| assert !palindrome?(not_pal) }
                  end
            
                def palindrome?(string)
                    string = string.gsub(/\W/, "").downcase
                    string == string.reverse
                end

Unit Test / Test Suite

assertion Örnekleri

  • assertEqual (a == b), assertNotEqual (a != b)
  • assertTrue (is True), assertFalse (is False)
  • assertIs (a is b), assertIsNot (a is not b)
  • assertIsNone (is None), assertIsNotNone (is not None)
  • assertIn (a in b), assertNotIn (a not in b)
  • assertIsInstance (isinstance(a, b)), assertNotIsInstance (not isinstance(a, b))
